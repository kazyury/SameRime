  def update_gui(list)
    @list=list
    @usersLV.clearItems
    list.each{|user_info|
      data=SameRime::Report.new(user_info)
      case data.status
        when "0" ; status="ACTIVE"
        when "1" ; status="STOPPED"
        else     ; status="UNKNOWN"
      end
      @usersLV.addItem([data.name,status,data.ipaddress,data.port])
    }
  end
