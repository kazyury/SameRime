  def initialize(controller)
    @controller=controller
    @server=TCPServer.open(@controller.config[:StatusManagerPort])
    @interval=@controller.config[:Interval]
    @status={}
    @logger=Log4r::Logger['default_logger']
    Thread.new{
      while true
        sleep @interval
        clean_up
      end
    }
  end
